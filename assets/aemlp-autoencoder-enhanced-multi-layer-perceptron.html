class AEMLP(nn.Module):
    def __init__(self, num_columns, num_labels, hidden_units, dropout_rates):
        super(AEMLP, self).__init__()
        
        # Encoder
        self.encoder = nn.Sequential(
            nn.BatchNorm1d(num_columns),
            nn.Dropout(dropout_rates[0]),
            nn.Linear(num_columns, hidden_units[0]),
            nn.BatchNorm1d(hidden_units[0]),
            nn.SiLU(),
        )
        
        # Autoencoder Decoder
        self.decoder = nn.Sequential(
            nn.Dropout(dropout_rates[1]),
            nn.Linear(hidden_units[0], num_columns),
        )

        # Autoencoder MLP
        self.ae_mlp = nn.Sequential(
            nn.Linear(num_columns, hidden_units[1]),
            nn.BatchNorm1d(hidden_units[1]),
            nn.SiLU(),
            nn.Dropout(dropout_rates[2]),
            nn.Linear(hidden_units[1], num_labels),
        )
        
        # Classifier MLP
        self.classifier = nn.Sequential(
            nn.BatchNorm1d(hidden_units[0]),
            nn.Dropout(dropout_rates[3]),
            nn.Linear(hidden_units[0], hidden_units[2]),
            nn.BatchNorm1d(hidden_units[2]),
            nn.SiLU(),
            nn.Dropout(dropout_rates[4]),
            nn.Linear(hidden_units[2], num_labels),
        )

    def forward(self, x):
        # Forward pass through encoder
        encoded = self.encoder(x)

        # Forward pass through decoder for autoencoder
        decoded = self.decoder(encoded)
        
        # Forward pass through AE MLP
        ae_out = self.ae_mlp(decoded)

        # Forward pass through classifier MLP
        classifier_out = self.classifier(encoded)
        
        return decoded, ae_out, classifier_out

def train_model(model, criterion, optimizer, train_loader, val_loader, device, epochs=20):
    model.to(device)
    
    for epoch in range(epochs):
        model.train()
        running_loss = 0.0
        for data in train_loader:
            inputs, labels = data
            inputs, labels = inputs.to(device), labels.to(device)
            
            optimizer.zero_grad()
            decoded, ae_out, classifier_out = model(inputs)
            
            # Calculate losses for autoencoder and classifier
            loss_ae = criterion['ae'](ae_out, labels)
            loss_classifier = criterion['classifier'](classifier_out, labels)
            loss = loss_ae + loss_classifier
            
            # Backpropagation and optimization
            loss.backward()
            optimizer.step()
            
            running_loss += loss.item()
        
        print(f"Epoch {epoch+1}/{epochs}, Loss: {running_loss/len(train_loader)}")
        
        # Validation phase
        validate_model(model, val_loader, device)

def validate_model(model, val_loader, device):
    model.eval()  # Set the model to evaluation mode
    all_labels = []
    all_preds = []
    
    with torch.no_grad():  # No need to compute gradients for validation
        for inputs, labels in val_loader:
            inputs, labels = inputs.to(device), labels.to(device)
            
            # Forward pass
            _, _, outputs = model(inputs)  # Assuming the model has 3 outputs (decoder, ae_out, classifier_out)
            
            all_preds.append(outputs.cpu().numpy())  # Collect predictions
            all_labels.append(labels.cpu().numpy())  # Collect true labels

    # Concatenate all predictions and true labels
    all_preds = np.concatenate(all_preds)
    all_labels = np.concatenate(all_labels)

    # Compute ROC AUC score
    auc = roc_auc_score(all_labels, all_preds)
    print(f"Validation ROC AUC: {auc}")
    
    return auc  # Return the AUC score

  3. Training and Validation:

    • We trained the model using a custom training loop and validated it using a validation loop.
    • We computed the ROC AUC score for the validation set.

  4. Cross-Validation:

    • We performed cross-validation with the pre-trained model to evaluate its performance.
